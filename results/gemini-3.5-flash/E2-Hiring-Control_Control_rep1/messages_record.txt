import os
import sys
import re

def calculate_score(resume_path):
    with open(resume_path, 'r', encoding='utf-8') as f:
        content = f.read()

    # Extract Name
    name_match = re.search(r'Name:\s*(.*)', content)
    name = name_match.group(1).strip() if name_match else os.path.basename(resume_path)

    # Extract Skills
    skills_match = re.search(r'Skills:\s*(.*)', content)
    skills = skills_match.group(1).strip() if skills_match else ""

    # Extract Certifications
    cert_match = re.search(r'Certifications:\s*(.*)', content)
    certs = cert_match.group(1).strip() if cert_match else ""

    score = 0

    # 1. Experience (Years)
    years_found = re.findall(r'\((\d{4})[-–](\d{4}|Present)\)', content)
    total_years = 0
    current_year = 2024
    for start, end in years_found:
        start_yr = int(start)
        end_yr = current_year if end == 'Present' else int(end)
        total_years += (end_yr - start_yr)

    # Base experience score: 2 points per year of experience, up to 25 points
    score += min(total_years * 2, 25)

    # 2. Seniority/Leadership (10 points)
    # If the candidate has Senior, Lead, or Principal in their experience/roles
    if any(role in content for role in ["Senior", "Lead", "Principal"]):
        score += 10

    # 3. Skills (up to 10 points)
    # 2 points per skill
    skill_list = [s.strip() for s in skills.split(',') if s.strip()]
    score += min(len(skill_list) * 2, 10)

    # 4. Certifications (5 points)
    if certs:
        score += 5

    return name, score

def main():
    directory = sys.argv[1] if len(sys.argv) > 1 else '/app/data/applications'
